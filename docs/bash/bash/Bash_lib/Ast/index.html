type command =
| SimpleCommand of string * expr list
| Compound of compound
| VarAssignment of arr
| Expression of expr
and arr =
| SimpleVar of var * expr
| Array of string * expr list
| AssociativeArray of string * (string * expr) list
and pipeline =
| CommandList of command list
| Pipe of command list
| AndList of pipeline list
| OrList of pipeline list
and const =
| Int of int
| String of string
| Bool of bool
and const_lst = const list
and expr =
| Const of const
| Binop of binop
| Var of var
| Assignment of var * expr
| SingleQuotes of string
| DoubleQuotes of expr list
| BraceExpansion of expr list
| ParameterExpansion of parameter_expansion
| ArithmeticExpansion of expr
| CommandSubstitution of command
| Redirect of redirect
and binop =
| Mul of expr * expr
| Add of expr * expr
| Sub of expr * expr
| Div of expr * expr
| Eq of expr * expr
| Ne of expr * expr
| Gt of expr * expr
| Ge of expr * expr
| Lt of expr * expr
| Le of expr * expr
and var = string * string
and compound =
| IfElse of expr * command list * command list option
| While of expr * command list
| For of expr * expr * expr * command list
| ForIn of expr * expr list * command list
| Case of expr * (expr list * command list) list
and parameter_expansion =
| Offset of var * expr
| OffsetLen of var * expr * expr
| Length of var
| ReplFirst of var * expr * expr
| ReplAll of var * expr * expr
| ReplBeg of var * expr * expr
| ReplEnd of var * expr * expr
| RemShortFromBeg of var * expr
| RemLargFromBeg of var * expr
| RemShortFromEnd of var * expr
| RemLargFromEnd of var * expr
| UseDefValue of var * expr
| SetDefValue of var * expr
| UseAlterValue of var * expr
and redirect =
| RedirectInput of expr
| RedirectOutput of expr
| AppendRedirOutput of expr
and funcn = string * command list
